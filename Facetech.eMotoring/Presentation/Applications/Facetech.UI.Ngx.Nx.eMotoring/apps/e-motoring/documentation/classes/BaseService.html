  protected handleError(error: any): Observable<any> {
    let applicationError: any = error.headers.get('Application-Error');

    // Either an applicationError in the header or a model error in the body
    if (applicationError) {
      this.errorNotification(applicationError, 'Application Error');
      return Observable.throw(applicationError);
    }

    let modelStateErrors: string = '';
    let serverError: any = error.json();

    if (!serverError.type) {
      for (var key in serverError) {
        if (serverError[key]) {
          modelStateErrors += serverError[key] + '\n';
        }
      }
    }

    modelStateErrors = modelStateErrors = '' ? null : modelStateErrors;
    this.errorNotification(applicationError, 'Model State Error');

    return Observable.throw(modelStateErrors || 'Server error');
  }

  protected notify(message: IMessage): void {
    //this.notificationService.notify(message);
  }

  private errorNotification(error: any, detail: string): void {
    this.notify(Message.create(MessageType.Error, detail, error));
  }
}
