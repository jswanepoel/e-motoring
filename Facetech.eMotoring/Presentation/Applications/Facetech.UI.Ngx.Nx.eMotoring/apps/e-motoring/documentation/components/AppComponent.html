import { Component, HostListener, Injector } from '@angular/core';
import { Subscription } from 'rxjs';
import { AppInjectorService } from './serivces/app-injector.service';
import { LoginService } from './serivces/login.service';

@Component({
  selector: 'facetech-root',
  templateUrl: './app.component.html',
  styleUrls: ['./app.component.css']
})

export class AppComponent {

  title = 'e-motoring';
  activated: boolean = true;
  public subscription: Subscription;
  private height: number = window.innerHeight;
 
  private status: boolean;



  public constructor(private injector: Injector) {
    AppInjectorService.setInjector(this.injector);
    document.documentElement.style.setProperty('--div-size', (this.height-310) + 'px');
  }

  @HostListener('window:resize', ['$event'])
  onResize(event) {
    this.height = event.target.innerHeight;
    document.documentElement.style.setProperty('--div-size', (this.height - 310) + 'px');
  }


  public calculatePreviewPageHeight(): number {
   

    return this.height - 174;
  }

}
