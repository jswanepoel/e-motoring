import { Component, OnInit, DebugElement } from '@angular/core';
import { SelectItem } from 'primeng/api';
import { ContactService } from './Service/contact.service';
import { Contact, MyClass } from './domain/contact';
import { By } from '@angular/platform-browser';


@Component({
  selector: 'ngx-contact-details',
  templateUrl: './contact-details.component.html',
  styleUrls: ['./contact-details.component.css']
})
export class ContactDetailsComponent implements OnInit {


  priority: SelectItem[];
  type: SelectItem[];
  contacts: Contact[];
  new: Contact;
  primaryOnly: boolean = true;
  total: number = 1;



  constructor(private contactService: ContactService) { }

  ngOnInit() {
    this.contactService.getContacts().then(contact => this.contacts = contact);
  


    this.priority = [
      { label: 'Please select', value: null },
      { label: 'Primary', value: 'Primary' },
      { label: 'Alternative', value: 'Alternative' }

    ];

    this.type = [
      { label: 'Please select', value: null },
      { label: 'Email', value: 'Email' },
      { label: 'Phone', value: 'Phone' }

    ];
  }

  add() {
    this.primaryOnly = false;
    this.contacts.push(new MyClass(this.total++));
    
 
  }

  updateTable(rowData)
  {

  }


  removeContact(rowData)
  {
    console.log(rowData.id);
    if (this.contacts.length != 1) {
      this.contacts.splice(this.contacts.indexOf(rowData), 1);
      this.total--;
    }
    else { this.primaryOnly = true; this.total = 1; }
  }


}
