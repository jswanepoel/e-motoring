import { Component, ElementRef, Input, OnInit, ViewChild, ViewChildren, OnDestroy } from '@angular/core';
import { FormGroup } from '@angular/forms';
import { isNullOrUndefined } from 'util';
import { DependentAttribute, DependentOption, QuestionDefinition, Condition } from '../question-base.component';
import { trigger, transition, animate, style } from '@angular/animations'
import { of, Subscription, Observable, Observer, Subject } from 'rxjs';


@Component({
  selector: 'ngx-question',
  templateUrl: './dynamic-question.component.html',
  styleUrls: ['./dynamic-question.component.css'],
  animations: [
    trigger('slideInOutQuestion', [
      transition(':enter', [
        style({ transform: 'translateX(-100%)', opacity: 0 }),
        animate('0.5s', style({ transform: 'translateX(0)', opacity: 1 }))
      ]), transition(':leave', [
        style({ transform: 'translateX(0)', opacity: 1 }),
        animate('0.5s', style({ transform: 'translateX(-100%)', opacity: 0 }))
      ])
    ]),
    trigger('slideInValidation', [
      transition(':enter', [
        style({ maxHeight: '0px', opacity: 0 }),
        animate('0.8s', style({ maxHeight: '100%', opacity: 1 }))
      ]), transition(':leave', [
        style({ maxHeight: '100%', opacity: 1 }),
        animate('0.1s', style({ maxHeight: '0px', opacity: 0 }))
      ])
    ])
  ]
})

export class DynamicQuestionComponent implements OnInit, OnDestroy {
   

  @Input() question: QuestionDefinition;
  @Input() form: FormGroup;
  @ViewChild('questionRef', { read: ElementRef }) questionRef: ElementRef;
  @Input() attributes: any = {};
  private _isLoading: boolean = false;

  @Input() set isLoading(isLoading: boolean) {
    this._isLoading = isLoading;
  }

  valueSubscription: Subscription;

  disabled: boolean = false;
  hidden: boolean = false;

  get isValid() { return this.form.controls[this.question.key].valid; };

  public ngOnInit() {
    this.toggleHide(this.question.hidden);

    this.valueSubscription = this.question.value.subscribe(
      x => { this.form.get(this.question.key).setValue(x, { onlySelf: true }) }
    );
    if (!isNullOrUndefined(this.question.dependentOptions)) {
      this.question.dependentOptions.forEach(dependentOption => {
        this.form.get(dependentOption.dependOnId).valueChanges.forEach((val: any) => {
          isNullOrUndefined(val) ? this.setAttributes(dependentOption, 'undefined') : this.setAttributes(dependentOption, isNullOrUndefined(val.value) ? val : val.value);
        });
      });
    }
  }

  ngOnDestroy(): void {
    this.valueSubscription.unsubscribe();
  }

  setAttributes(depOption: DependentOption, val: any) {
    depOption.attributes.forEach(attr => {
      if (attr.conditions.some(x => x.conditionVal == val)) {
        attr.conditions.forEach(condition => {
          if (condition.conditionVal == val) {
            this.conditionMatch(condition, attr);
          }
        })
      }
      else {
        this.conditionFail(attr);
      }
    });
  }

  conditionMatch(cond: Condition, attr: DependentAttribute) {
    (attr.attrName == 'disabled') ? this.toggleDisable(cond.onMatchSetTo) :
      (attr.attrName == 'hidden') ? this.toggleHide(cond.onMatchSetTo)
        : attr.isBoundAttribute ? this.setBoundAttribute(attr, cond)
          : this.setNativeElementAttribute(attr, cond);
  }

  conditionFail(attr: DependentAttribute) {
    attr.conditions.forEach(cond => {
      let hasFailVal = !isNullOrUndefined(cond.onFailSetTo);
      (attr.attrName == 'disabled') ? this.toggleDisable(hasFailVal ? cond.onFailSetTo : !cond.onMatchSetTo) :
        (attr.attrName == 'hidden') ? this.toggleHide((hasFailVal ? cond.onFailSetTo : !cond.onMatchSetTo))
          : attr.isBoundAttribute ?
            hasFailVal ? this.attributes[attr.attrName] = cond.onFailSetTo : delete this.attributes[attr.attrName]
            : hasFailVal ? this.questionRef.nativeElement[attr.attrName] = cond.onFailSetTo : delete this.questionRef.nativeElement[attr.attrName];
    });
  }

  getAttrValFromList(val: string) {
    try {
      return this.question.questionOptions.multiOptions.find(item => ((item.value == val) || (item.label == val)));
    } catch{
      return undefined;
    }

  }

  toggleHide(setTo) {
    //this.toggleDisable(setTo);//disable and hide to avoid validation errors from hidden values.
    
    this.hidden = setTo;
    this.form.controls[this.question.key].reset();
    this.hidden ? this.form.controls[this.question.key].disable() : this.form.controls[this.question.key].enable();
  }

  toggleDisable(val: boolean) {
    if (val) {
      this.form.controls[this.question.key].disable();
    }
    else {
      this.form.controls[this.question.key].enable();
    }
  }

  setBoundAttribute(attr, cond) {
    this.attributes[attr.attrName] = this.getAttrValFromList(cond.onMatchSetTo) ? this.getAttrValFromList(cond.onMatchSetTo).value : cond.onMatchSetTo;
  }

  setNativeElementAttribute(attr, cond) {
    this.questionRef.nativeElement[attr.attrName] = this.getAttrValFromList(cond.onMatchSetTo) ? this.getAttrValFromList(cond.onMatchSetTo).value
      : cond.onMatchSetTo;
  }
}
