      transition(':enter', [

  public hamburger: boolean;

  @HostListener('window:resize', ['$event'])
  sizeWindow(event) {
    this.wt = event.target.innerHeight;
    this.sizeMenu(this.wt);
  }

  public sizeMenu(wt: any): void {
    if (wt >= 800) {
      this.hamburger = true;

    } else if (wt < 800) {
      this.hamburger = false;
    }
  }

  public sidePanelCalc() {
    return (this.wt - 57.5) + 'px';
  }
