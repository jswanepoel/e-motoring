  getIdMask(event) {

    if (event.value == "New NIC") {
      this.maskId = "";
      this.userform.controls.identificationNumber.setValue("");
      this.maskId = this.ValidationService.getIdMask();
    }
    else if (event.value == "NIC") {
      this.maskId = "";
      this.userform.controls.identificationNumber.setValue("");
      this.maskId = this.ValidationService.getIdMaskAlt();
    }
    else if (event.value == "BusinessRegistartionNo") {
      this.maskId = "";
      this.userform.controls.identificationNumber.setValue("");
      this.maskId = this.ValidationService.getIdBusinessCode();
    }
    else if (event.value == "passportNumber") {
      this.maskId = "";
      this.userform.controls.identificationNumber.setValue("");
      this.maskId = this.ValidationService.getIdBusinessCode();
    }
    else {
      this.question[0] = true;
    }
  }
